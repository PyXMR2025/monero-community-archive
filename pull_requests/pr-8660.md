This can be upgraded with old school MPL:
```c++
template<typename T>
static int type_index_of()
{
    using vtype = boost::mpl::vector<Types...>;
    using elem = typename boost::mpl::find<vtype, T>::type;
    using begin = typename boost::mpl::begin<vtype>::type;
    return 1 + boost::mpl::distance<begin, elem>::value;
}
```
