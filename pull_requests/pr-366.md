## hinto-janai | 2025-01-24T22:57:58+00:00
> it may actually be better to atomically modify the required fields instead of swapping the whole struct

@SyntheticBird45 all the fields in `ContextToVerifyBlock` must be updated together atomically, this is not possible. Updating them separately would run into the [ABA problem](https://en.wikipedia.org/wiki/ABA_problem) e.g. `height` is 1 but `hash` is still block 0.

Right now the best thing you can do is either:
1. Atomically update the largest value supported by the architecture (practically it's 64 bits if you want to stay portable)
2. Atomically update the pointer (64 bits) to the data structure (`arc_swap` does this and handles the problems that pop up, e.g. deallocatation)

Here is some other work on this topic if you wanted to learn more:
- https://www.cs.otago.ac.nz/cosc440/readings/hazard-pointers.pdf
- https://docs.rs/crossbeam/latest/crossbeam/epoch/index.html
- https://www.youtube.com/watch?v=fvcbyCYdR10

fn main() {
    // generate 100 transaction hashes
    let v1: Vec<[u8; 32]> = (0..100).map(|_| rand::random()).collect();

    // create a clone, but ordered differently
    let mut v2 = v1.clone();
    v2.shuffle(&mut thread_rng());

    // both vecs contain the same hashes but are in different ordering
    assert_eq!(v1.len(), v2.len());
    assert!(v1 != v2);

    // ordering function errors
    assert!(order_transactions(v1, v2).is_err());
}

fn order_transactions(v1: Vec<[u8; 32]>, mut v2: Vec<[u8; 32]>) -> Result<(), ()> {
    for (i, tx_hash) in v1.iter().enumerate() {
        if tx_hash != &v2[i] {
            let Some(at_index) = v2[i..].iter().position(|tx| tx == tx_hash) else {
                return Err(());
            };

            v2.swap(i, at_index);
        }
    }

    Ok(())
}

fn test(
    v1: Vec<[u8; 32]>,
    v2: Vec<[u8; 32]>,
    f: fn(v1: &Vec<[u8; 32]>, v2: &mut Vec<[u8; 32]>),
    name: &str,
) {
    let now = Instant::now();
    for _ in 0..1000000 {
        f(&b(v1.clone()), &mut b(v2.clone()));
    }
    println!("{name}: {}", now.elapsed().as_secs_f32());
}

fn full_clone(v1: &Vec<[u8; 32]>, v2: &mut Vec<[u8; 32]>) {
    if v1 != v2 {
        *v2 = v1.clone();
    }
}

fn iter_branch_clone(v1: &Vec<[u8; 32]>, v2: &mut Vec<[u8; 32]>) {
    for (h1, h2) in v1.iter().zip(v2) {
        if h1 != h2 {
            *h2 = *h1;
        }
    }
}

fn iter_position_swap(v1: &Vec<[u8; 32]>, v2: &mut Vec<[u8; 32]>) {
    for (i, tx_hash) in v1.iter().enumerate() {
        if tx_hash != &v2[i] {
            if let Some(at_index) = v2[i..].iter().position(|tx| tx == tx_hash) {
                v2.swap(i, at_index);
            }
        }
    }
}
