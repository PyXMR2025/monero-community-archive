`cmake ../.. \`
`-DCMAKE_TOOLCHAIN_FILE=$NDK/build/cmake/android.toolchain.cmake \`
`-DANDROID_ABI=armeabi-v7a \`
`-DANDROID_NATIVE_API_LEVEL=android-22 \`
`-DWITH_CUDA=OFF \`
`-DWITH_OPENCL=OFF \`
`-DWITH_HWLOC=OFF \`
`-DWITH_TLS=OFF \`
`-DUV_LIBRARY=../../../libuv-1.37.0/build/ARMv7a/libuv_a.a \`
