This should be equivalent (it passes these tests):

```c++
void buffer::append(const void* data, size_t sz)
{
  size_t moved = 0;
  const size_t capacity = storage.capacity();
  const size_t avail = capacity - storage.size();

  if (sz > avail || (size() <= 4096 && offset > 4096 * 16 && offset >= capacity / 2))
  {
    if (size() + sz > capacity)
      storage.reserve((((size() + sz) * 3 / 2) + 4095) & ~4095);

    memmove(storage.data(), storage.data() + offset, storage.size() - offset);
    moved = offset;
    offset = 0;
  }

  const size_t current = storage.size() - moved;
  storage.resize(current + sz);
  memmove(storage.data() + current, data, sz);
}
```
